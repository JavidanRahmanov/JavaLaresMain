package Tasks.lesson7;

import java.util.Scanner;
import java.util.Random;
public class RandomShipLocator {
    public static void ShipDeterminator(String[][] matrix) {

        Scanner sc = new Scanner(System.in);

        String sorgu = " ";
        while (!sorgu.equals("xeyr")) {
            System.out.println("Daxil etmek istediyiniz gemileri secin: ");
            System.out.println("1-Aircraft Carrier(5 cells)");
            System.out.println("2-Battleship(4 cells)");
            System.out.println("3-Submarine(3 cells)");
            System.out.println("4-Destroyer(3 cells)");
            System.out.println("5-Patrol Boat(2 cells)");
            int choose = sc.nextInt();
            sc.nextLine();
            int numberOfReplacements;
            switch (choose) {
                case 1:
                    numberOfReplacements = 5;
                    break;
                case 2:
                    numberOfReplacements = 4;
                    break;
                case 3:
                    numberOfReplacements = 3;
                    break;
                case 4:
                    numberOfReplacements = 3;
                    break;
                case 5:
                    numberOfReplacements = 2;
                    break;
                default:
                    System.out.println("Yanlış seçim, doğru bir seçim edin.");
                    continue;
            }
            ShipLocator(matrix,"p",numberOfReplacements);
            System.out.println("Basqa Gemi elave etmek isterdiniz mi");
            sorgu = sc.nextLine().toLowerCase();
        }
    }

    public static void ShipLocator(String[][] matrix, String replacement, int count) {
        Random random = new Random();
        int rows = matrix.length;
        int cols = matrix[0].length;
        boolean replaced = false;

        while (!replaced) {

            int startRow = random.nextInt(rows);
            int startCol = random.nextInt(cols);

            boolean canReplaceHorizontally = (startCol + count <= cols);
            boolean canReplaceVertically = (startRow + count <= rows);

            if (canReplaceHorizontally && canReplaceVertically) {

                int direction = random.nextInt(2);
                if (direction == 0) {
                    for (int i = 0; i < count; i++) {
                        matrix[startRow][startCol + i] = replacement;
                    }
                } else {
                    for (int i = 0; i < count; i++) {
                        matrix[startRow + i][startCol] = replacement;
                    }
                }
                replaced = true;
            } else if (canReplaceHorizontally) {
                for (int i = 0; i < count; i++) {
                    matrix[startRow][startCol + i] = replacement;
                }
                replaced = true;
            } else if (canReplaceVertically) {

                for (int i = 0; i < count; i++) {
                    matrix[startRow + i][startCol] = replacement;
                }
                replaced = true;
            }
        }
    }
}
