package Tasks.lesson7;

import java.util.Scanner;


public class BattleshipWarGame {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("----------------------Welcome to Battleship War!-----------------------");
        String map1[][] = new String[10][10];
        String map2[][] = new String[15][20];
        String map3[][] = new String[20][20];
        String[][] newMap;
        System.out.println("Choose one of the maps: ");
        System.out.println("1-Green Hills");
        System.out.println("2-Perilous Cliffs");
        System.out.println("3-North Mountains");
        int choose = sc.nextInt();
        switch(choose){
            case 1: newMap = map1;
            break;
            case 2: newMap = map2;
            break;
            case 3: newMap = map3;
            break;
            default:
                System.out.println("Invalid choice. Defaulting to Green Hills.");
                newMap = map1;
                break;
        }
        for (int i = 0; i < newMap.length; i++) {
            char rowLabel = (char) ('A' + i);
            for (int j = 0; j < newMap[i].length; j++) {
                newMap[i][j] = rowLabel + Integer.toString(j + 1);
            }
        }
        RandomShipLocator.ShipDeterminator(newMap);
        for(int i = 0; i < newMap.length; i++){
            for(int j = 0; j < newMap[i].length; j++){
                System.out.print(newMap[i][j] + " ");
            }
            System.out.println();
        }
    }
}
